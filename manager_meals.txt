        document.addEventListener("DOMContentLoaded", function() {
            // ------------ bottom save ----------------
            const bottomsave= document.getElementById("bottomsave");
            
            //
            const foods = JSON.parse('{{ foods | tojson | safe }}');
            const id_food_methods = JSON.parse('{{methods_food | tojson | safe }}')

            // ----------- subgroups -> foods -----------
            
            const subgroups = document.querySelectorAll(".subgroup_select");
            
            subgroups.forEach(function(select){
                select.addEventListener("change", function(){
                    const subgroup_id = this.value;
                    const target_id = this.dataset.foodtarget;
                    const food_select = document.getElementById(target_id);
                    console.log("subgroup_id:", subgroup_id);
                    console.log("target_id:", target_id);
                    console.log("food_select:", food_select);
                    if (!food_select) {
                console.log("Select de alimento não encontrado para:", target_id);
                return;
            }

                    food_select.innerHTML = '<option value="" disabled selected>Escolha....</option>';
                    if (!subgroup_id || !foods[subgroup_id]) return;
                    foods[subgroup_id].forEach(function(food){
                        const option = document.createElement("option");
                        option.value = food.id;
                        option.textContent = food.name;
                        food_select.appendChild(option);
                    });
                });
            });

            // ---------------- foods -> methods ----------

            const food_selects = document.querySelectorAll(".food_select")

                food_selects.forEach(function(select){
                    select.addEventListener("change", function(){
                    const food_id = this.value;
                    const target_method = this.dataset.foodmethod;
                    const method_select = document.getElementById(target_method)
                    method_select.innerHTML = '<option value="" disabled selected>Metodo....</option>';
                    if (!id_food_methods[food_id]) return;
                    id_food_methods[food_id].forEach(function(method){
                        const option = document.createElement("option");
                        option.value = method.id;
                        option.textContent = method.name;
                        method_select.appendChild(option)
                    });
                });
            });
    });
